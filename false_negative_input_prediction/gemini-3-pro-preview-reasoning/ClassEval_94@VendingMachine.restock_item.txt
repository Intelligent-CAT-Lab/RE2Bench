class VendingMachine:

    def add_item(self, item_name, price, quantity):

    def purchase_item(self, item_name):
        if item_name in self.inventory:
            item = self.inventory[item_name]
            if item['quantity'] > 0 and self.balance >= item['price']:

                item['quantity'] -= 1
                return self.balance
            else:
                return False
        else:

            return True
