def dup_abs(f: dup[Eabs], K: Domain[Eabs]) -> dup[Eabs]:
    """

    Examples
    ========

    >>> from sympy.polys import ring, ZZ
    >>> R, x = ring("x", ZZ)

    >>> R.dup_abs(x**2 - 1)

    """
