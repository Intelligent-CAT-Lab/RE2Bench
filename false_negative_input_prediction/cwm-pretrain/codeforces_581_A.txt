        | expected_output | string | 期望输出

        | test_stats | list | 用于存放测试统计数据

        self.expected_output = expected_output
        self.output = None

            assert len(input) == 2, f'Input {input} is not a valid test input, if you want to set pre or post index, it should be a tuple or list with length 2'

            self.input_post_index = input[1]
            self.input = self.input_pre_index + self.input_post_index
            self.is_split_input = True

            self.input = input
            self.input_pre_index = None
            self.input_post_index = None
            self.is_split_input = False
        self.check_mode = check_mode
        self.resub = resub
        self.allow_comment = allow_comment
        self.test_stats = test_stats
        self.test_passed = False
