from sklearn.pipeline import Pipeline
from sklearn.tree import DecisionTreeClassifier
from sklearn.svm import LinearSVC
from sklearn.gaussian_process import GaussianProcessClassifier
from sklearn.ensemble import RandomForestClassifier
from sklearn.datasets import make_classification
from sklearn.model_selection import cross_val_score

X, y = make_classification(random_state=42, n_samples=10)
pipe = Pipeline([
    ("classify", RandomForestClassifier(random_state=42)),
    ("truncate", "trunc"),
    ("original", GaussianProcessClassifier(1.0, (1e-10, 1e5), 1.0, 1, "gaussian"))
])
print("P long")
pipe.set_params(truncate="".join(["what", "if", "this", "is", "super", "long", "and", "we", "need", "to", "truncate", "it", "to", "just", "show", "the", "prefix"]))
cross_val_score(pipe, X, y, cv=2)

print("P dict")
pipe.set_params(original={"A": 1, "B": 2})
cross_val_score(pipe, X, y, cv=2)

print("P int")
pipe.set_params(original=42)
cross_val_score(pipe, X, y, cv=2)

print("P obj")
pipe.set_params(original=GaussianProcessClassifier(1.0, (1e-10, 1e5), 1.0, 1, "gaussian"))
cross_val_score(pipe, X, y, cv=2)

print("P list")
pipe.set_params(original=[1,2,3])
cross_val_score(pipe, X, y, cv=2)

print("P default")
pipe.set_params(original=DecisionTreeClassifier(random_state=42))
cross_val_score(pipe, X, y, cv=2)

print("P Custom Repr")
pipe.set_params(original=LinearSVC(dual=True, C=1.0))
cross_val_score(pipe, X, y, cv=2)

print("P Changed params")
pipe.set_params(original=RandomForestClassifier(42))
cross_val_score(pipe, X, y, cv=2)
