# Function to get input values for XXX placeholders from a complex structure
def get_input_values(output, structure):
    # Initialize variables to store input values
    input_values = {}
    
    # Parse output to extract values
    output_values = output.strip().strip("{}").split(", ")
    
    # Parse structure to extract keys and types
    structure_values = structure.strip().strip("{}").split(", ")
    
    # Iterate through structure values
    for structure_value in structure_values:
        key, value_type = structure_value.split(": ")
        
        # Check if value type is string and output value is between double quotes
        if value_type.startswith("str") and output_values[0].startswith('"'):
            input_values[key] = output_values[0].strip('"')
        # Check if value type is integer
        elif value_type.startswith("int"):
            input_values[key] = int(output_values[0])
        # Check if value type is float
        elif value_type.startswith("float"):
            input_values[key] = float(output_values[0])
        # Check if value type is list
        elif value_type.startswith("list"):
            input_values[key] = [output_values[i].strip('"') for i in range(1, len(output_values))]
        # Check if value type is dictionary
        elif value_type.startswith("dict"):
            input_values[key] = {}
            for i in range(1, len(output_values)):
                key, value = output_values[i].split(": ")
                input_values[key][key.strip('"')] = value.strip('"')
    
    return input_values
