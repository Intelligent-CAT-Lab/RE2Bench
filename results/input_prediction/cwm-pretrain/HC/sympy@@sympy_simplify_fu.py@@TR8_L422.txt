def parse_num(maybe_num):
    try:
        return int(maybe_num)
    except ValueError:
        return -1

def extract_suffixes(iter, prefix):
    p_len = len(prefix)
    for p in iter:
        yield p.name[p_len:]


def find_suffixes(root, prefix):
    return extract_suffixes(find_prefixed(root, prefix), prefix)

def make_numbered_dir(root, prefix):
    for i in range(10):
        # try up to 10 times to create the folder
        max_existing = _max(map(parse_num, find_suffixes(root, prefix)), default=-1)
        new_number = max_existing + 1
        new_path = root.joinpath("{}{}".format(prefix, new_number))
        try:
            new_path.mkdir()
        except Exception:
            pass
        else:
            _force_symlink(root, prefix + "current", new_path)
            return new_path
    else:
        raise EnvironmentError(
            "could not create numbered dir with prefix "
            "{prefix} in {root} after 10 tries".format(prefix=prefix, root=root)
        )

from six import string_types, binary_type, text_type

def normalize_nodes(sep):
    """Normalize the nodes to split strings in nodes"""
    if isinstance(sep, string_types):
        sep = sep.encode('utf-8')

    if not isinstance(sep, binary_type):
        raise TypeError("expected type str, unicode, or binary_type; "
            "got {0!r}".format(type(sep)))

    # Ensure we do not have an empty separator
    if not sep:
        raise ValueError("expected non-empty separator")

    # Cache Sep
    sep = b'/' + (sep.lstrip(b'/') or b'/') + b'/'
    return sep


def normalize_pathsep(pathsep):
    """Normalize the path separator to split PATH like variables."""
    if isinstance(pathsep, string_types):
        pathsep = pathsep or ':'

        if len(pathsep) != 1:
            raise ValueError("expected single character delimiter")

        pathsep = pathsep.encode('utf-8')

    else:
        pathsep = os.path.pathsep.encode('utf-8')

    return pathsep

class ExecutionContext(object):
    def __init__(self, **kw):
        self.keys = kw.keys()
        self._data = kw.copy()
        self._original = kw.copy()

    def set_value(self, name, value, excinfo=None, assign=False):
        # ^^^ this is called before the variable is really created (if at
        # all), in the setup-phase. The value is only put into the public
        # dictionary by `setoutcomedata` (in outcome.py)
        old_value = None
        if name in self._data:
            old_value = self._data[name]
        if assign:
            assert excinfo is not None
            assert old_value is not None
        if not assign:
            self._data[name] = value
        return (old_value, assign)

    def get_value(self, name):
        if name in self._data:
            return self._data[name]
        elif self._original:
            return self._original[name]
        return None

    def clear(self):
        if self._data:
            self._data.clear()
            self.keys.clear()
