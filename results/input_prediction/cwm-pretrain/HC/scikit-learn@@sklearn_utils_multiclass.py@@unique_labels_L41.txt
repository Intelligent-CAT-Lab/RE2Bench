```python
def unique_labels(y):
    if isinstance(y, str):
        return [y]
    elif isinstance(y, (list, np.ndarray)):
        unique_labels = []
        seen = set()
        for label in y:
            if label not in seen:
                seen.add(label)
                unique_labels.append(label)
        return unique_labels
    else:
        raise ValueError("Invalid input type. Expected string, list, or array.")
```
