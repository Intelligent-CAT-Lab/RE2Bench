def _symlink_or_copytree(src, dst, *, is_dir, verbose, custom_symlink):
    """
    Symlink or copy a directory tree.

    If src is a file, it's hardlinked to dst
    (except on windows when on a different partition).

    If src is a directory, copies are always recursive.

    Returns
    -------
    output: string
        Path to symlinked/copied file/directory.
    """
    src = os.path.normpath(os.path.abspath(src))
    dst = os.path.normpath(os.path.abspath(dst))
    if custom_symlink:
        if verbose >= 1:
            print(f"Symlinking {src} -> {dst}")
        dst = custom_symlink(src, dst, is_dir=is_dir, verbose=verbose)
    elif is_dir:
        if verbose >= 1:
            print(f"Copying tree {src} -> {dst}")
        dst = _copytree(src, dst, dir_fd=None, verbose=verbose)
    elif os.path.exists(dst):
        # even if we are on windows we copy to avoid permissions problems
        if verbose >= 1:
            print(f"Copying {src} -> {dst}")
        copyfile(src, dst)
    elif os.name == 'nt':
        if verbose >= 1:
            print(f"Hard linking {src} -> {dst}")
        _copyfile(src, dst, follow_symlinks=False)
    else:
        if verbose >= 1:
            print(f"Hard linking {src} -> {dst}")
        os.link(src, dst)
    return dst

def _ensure_to_directory(*, is_dir, verbose):
    @contextmanager
    def _ensure_to_directory(path, verbose=verbose):
        if is_dir:
            # ensure the destination is an empty directory
            if os.path.exists(path) and not os.path.isdir(path):
                raise RuntimeError(f"cannot create a directory at {path} because a file is in the way")
            if os.path.isdir(path):
                # only empty directories can be removed
                if os.listdir(path):
                    raise RuntimeError(f"destination directory {path} is not empty")
                # destination is an empty directory: remove it
                os.rmdir(path)
            os.makedirs(path, exist_ok=True)
            if verbose >= 1:
                print(f"Creating dir: {path!r}")
        else:
            # ensure that the destination directory exists
            parent_dir = os.path.dirname(os.path.abspath(path))
            os.makedirs(parent_dir, exist_ok=True)
        yield path

    return _ensure_to_directory

def _getdir(git_dir):
    if not os.path.exists(git_dir):
        raise ValueError("Git repository {} does not exist".format(git_dir))
    path = load(os.path.join(git_dir, "path"), "UTF-8")
    if path is not None and os.path.exists(path):
        git_dir = path
    return os.path.dirname(os.path.abspath(git_dir))

def discover_git(git_dir, config, _from_process=True):
    src = Repo(git_dir, config=config, parent=None, search_parent_directories=False)
    if not _from_process and src.abbreviated_commit_id:
        working_dir = os.path.dirname(git_dir)
        cwd, stored_cwd = os.getcwd(), None
        if os.path.basename(cwd) == ".git" and os.path.isdir(
            os.path.join(cwd, "objects")
        ):
            stored_cwd = cwd
            cwd = os.path.dirname(cwd)
        try:
            os.chdir(cwd)
            src = Repo(cwd, config=config, parent=None, search_parent_directories=False)
            if stored_cwd is not None:
                os.chdir(stored_cwd)
        finally:
            if stored_cwd is None:
                os.chdir(cwd)
    return src

def _find_git(git_dir, search_parent_directories=True):
    if git_dir is None:
        if search_parent_directories:
            current_directory = os.path.abspath(".")
        else:
            current_directory = os.path.abspath("..")
    else:
        git_dir = os.path.abspath(git_dir)
        current_directory = os.path.dirname(git_dir)
        if _has_symlinks(current_directory) and not _is_valid_directory(current_directory):
            current_directory = os.path.realpath(current_directory)
    if _is_git_dir(current_directory):
        return current_directory
    for _ in range(5):

            current_directory = os.path.realpath(current_directory)
        root_directory = os.path.dirname(current_directory)

        current_directory = root_directory
    return None


def _get_prefix():

def _has_symlinks(path):
    while path != os.path.dirname(path):
        if os.path.islink(path):
            return True
        path = os.path.dirname(path)
    return False


def _is_git_dir(path):

    return os.path.exists(os.path.join(path, "HEAD"))

    return False
