def extract_suffixes(iter, prefix):
    p_len = len(prefix)
    for p in iter:

def ensure_reset_dir(p):
    """
    Resets directory if it exists; if not does nothing.
    Returns True if a reset occurred.
    """
    try:
        p = p.resolve()
    except OSError:
        return False  # not a dir
    else:
        if p.is_dir():
            for d in p.iterdir():
                rmtree(d, ignore_errors=True)
            return True
        return False


def make_numbered_dir(root, prefix):
    """
    Create a new numbered directory, and then link prefix+current to it.

    Tries to generate unique numbers by checking the maximum number in root.
    Does not check any properties other than name: any directories named
    (prefix + str(num)) are assumed to be directories previously created by
    make_numbered_dir().

    Does not keep trying forever when collisions occur (i.e. when something
    outside of this function tries to create a directory while this function
    is executing), instead it gives up after a fixed number of attempts.

    Returns the path to the created directory.
    """
    for i in range(10):
        # try up to 10 times to create the folder
        max_existing = _max(map(parse_num, find_suffixes(root, prefix)), default=-1)
        new_number = max_existing + 1
        new_path = root.joinpath("{}{}".format(prefix, new_number))
        try:
            new_path.mkdir()
        except Exception:
            pass
        else:
            _force_symlink(root, prefix + "current", new_path)
            return new_path
    else:
        raise EnvironmentError(
            "could not create numbered dir with prefix "
            "{prefix} in {root} after 10 tries".format(prefix=prefix, root=root)
        )
