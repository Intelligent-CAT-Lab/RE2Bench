def gram_schmidt(V, method='mgs'):
    # V is a set of column vectors
    # gram_schmidt(V) returns the Q matrix of the QR decomposition
    # of the column vectors in V
    A, R = np.linalg.qr(V.T)
    R = np.triu(R)
    if method == 'mgs':
        Q = V.copy()
        for k in range(A.shape[1]):
            R[k,k] = np.linalg.norm(A[:,k])
            Q[:,k] = A[:,k] / R[k,k]
            for j in range(k+1,A.shape[1]):
                R[k,j] = Q[:,k].dot(A[:,j])
                A[:,j] = A[:,j] - R[k,j] * Q[:,k]
    elif method == 'cgs':
        A = V.copy()
        Q = A.copy()
        for k in range(A.shape[1]):
            v = A[:,k]
            for j in range(k):
                R[j,k] = Q[:,j].dot(v)
                v = v - R[j,k] * Q[:,j]
            R[k,k] = np.linalg.norm(v)
            Q[:,k] = v / R[k,k]
    else:
        raise ValueError("Method must be 'mgs' or 'cgs'")
    return Q, R
