def insertion_sort(data: List[Tuple[str, int]]) -> List[Tuple[str, int]]:
    for i in range(1, len(data)):
        key = data[i]
        j = i - 1
        while j >= 0 and key[0] < data[j][0]:
            data[j + 1] = data[j]
            j -= 1
        data[j + 1] = key
    return data
