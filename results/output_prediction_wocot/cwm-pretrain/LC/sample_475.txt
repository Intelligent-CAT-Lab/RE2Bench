import config  # pylint: disable=import-error
import threading
import time


def display_graph():
    graph = {
        'edges': [],
        'nodes': {},
        'layout': {
            'columns': 5,
            'width': 360,
            'height': 360,
            'shape': 360 / 5,
            'transform': (0, 0, 0, 360 / 5),
        }
    }
    while not graph['layout']['done']:
        config.graph_surface.clear()
        config.graph_surface.drawRect(0, 0, 360, 360, color=(0, 0, 0, 0))
        angle = 0
        for edge in graph['edges']:
            config.graph_surface.drawLine(
                graph['nodes'][edge[0]]['x'],
                graph['nodes'][edge[0]]['y'],
                graph['nodes'][edge[1]]['x'],
                graph['nodes'][edge[1]]['y'],
                color=(255, 255, 255, 255),
                width=2,
            )
        for node in graph['nodes']:
            config.graph_surface.drawCircle(
                graph['nodes'][node]['x'],
                graph['nodes'][node]['y'],
                graph['nodes'][node]['radius'],
                color=(255, 255, 255, 255),
            )
        angle += graph['layout']['shape']
        config.graph_surface.rotate(1, 0, 0, 360 / graph['layout']['columns'])
        time.sleep(0.01)
    config.graph_surface.clear()


def update_graph():
    config.graph_lock.acquire()
    config.graph_surface.rotate(1, 0, 0, 360 / config.graph_surface.width)
    config.graph_lock.release()
    time.sleep(0.01)


def setup_graph():
    config.graph_surface = config.Surface(
        (360, 360), pyautogui.pixel(0, 0), convert_alpha=255
    )
    config.graph_surface.set_alpha(255)
    config.graph_lock = threading.Lock()
    config.graph_thread = threading.Thread(target=display_graph)
    config.graph_thread.daemon = True
    config.graph_thread.start()

from collections import defaultdict
from dataclasses import dataclass
from typing import List, Dict, Tuple

import numpy as np
import pandas as pd


@dataclass
class _InputData:
    input_data: Dict[str, List[str]]


class _GraphOutput:
    def __init__(self, headers, values):
        self.headers = headers
        self.values = values

    def to_dataframe(self):
        return pd.DataFrame(self.values, columns=self.headers)

    def __str__(self):
        df = self.to_dataframe()
        return df.to_string(index=False)


def output_edge_list(
    input_data: _InputData,
    cooccur_value_threshold: int = 1,
    cooccur_entity_threshold: int = 5,
) -> _GraphOutput:
    """
    Summarize co-occurrence graph information from input data

    Args:
        input_data: Entity-Entity and Entity-Aspect dictionaries
        cooccur_value_threshold: Threshold for co-occurrence value
        cooccur_entity_threshold: Threshold for number of co-occurring entities

    Returns:
        A GraphOutput object containing the edge list and node labels for the co-occurrence graph
    """

    entity_entity_dict = input_data.input_data["entity-entity"]
    entity_aspect_dict = input_data.input_data["entity-aspect"]

    edge_list = defaultdict(int)

    for entity1 in entity_entity_dict:
        for entity2 in entity_entity_dict[entity1]:
            if entity2 not in entity_aspect_dict:
                continue
            if entity_entity_dict[entity1][entity2] > cooccur_value_threshold:
                edge_list[(entity1, entity2)] = entity_entity_dict[entity1][entity2]

    filtered_edges = [
        edge for edge in edge_list if edge_list[edge] >= cooccur_entity_threshold
    ]
    filtered_edges.sort(key=lambda x: edge_list[x], reverse=True)

    node_labels = {}
    for entity in entity_entity_dict:
        node_labels[entity] = 0
    for entity in entity_aspect_dict:
        node_labels[entity] = 1

    return _GraphOutput(["Edge", "Weight"], filtered_edges)


def to_dataframe(node_labels: Dict[str, int], edge_list: List[Tuple[str, str, int]]) -> pd.DataFrame:
    """
    Convert the node labels and edge list to a pandas DataFrame.

    Args:
        node_labels: A dictionary of node labels
        edge_list: A list of tuples, each tuple containing the source and target node, and the weight

    Returns:
        A pandas DataFrame containing the node labels and edge list
    """
    edge_df = pd.DataFrame(edge_list, columns=["Source", "Target", "Weight"])
    node_df = pd.DataFrame(list(node_labels.items()), columns=["Node", "Label"])
    df = edge_df.merge(node_df, left_on="Source", right_on="Node")
    df = df.merge(node_df, left_on="Target", right_on="Node", suffixes=("_source", "_target"))
    df.drop("Node", axis=1, inplace=True)
    return df


def to_newick(node_labels: Dict[str, int], edge_list: List[Tuple[str, str, int]]) -> str:
    """
    Convert the node labels and edge list to a Newick tree string.

    Args:
        node_labels: A dictionary of node labels
        edge_list: A list of tuples, each tuple containing the source and target node, and the weight

    Returns:
        A Newick tree string
    """
    df = to_dataframe(node_labels, edge_list)
    tree = df["Source"].map(lambda x: "(" + x + ")") + df["Target"].map(lambda x: "(" + x + ")") + df["Weight"].map(lambda x: ":" + str(x))
    return tree.str.cat(sep="")


def main():
    data = {
        "entity-entity": {
            "A": {"B": 5, "C": 3},
            "B": {"A": 5, "C": 2},
            "C": {"A": 3, "B": 2, "D": 7},
            "D": {"C": 7},
        },
        "entity-aspect": {"A": 1, "B": 2, "C": 3, "D": 4},
    }

    input_data = _InputData(input_data=data)
    graph_output = output_edge_list(input_data)

    print(graph_output)
    df = to_dataframe(graph_output.node_labels, graph_output.edge_list)
    print(df)


if __name__ == "__main__":
    main()

def visualize(df: pd.DataFrame) -> None:
    """
    Function to visualize the dataframe using a treeview.

    Parameters:
        df (pd.DataFrame): The dataframe to visualize.
    """
    try:
        # Get the columns and data from the dataframe
        columns = df.columns.tolist()
        data = df.values.tolist()

        # Initialize a new MainApplication
        app = MainApplication()

        # Create nodes for the treeview
        nodes = []
        for i, row in enumerate(data):
            node = NodeData(
                children=None,
                column=0,
                data=None,
                key=str(i),
                label=str(i + 1),
                node_data=Data(
                    value=row,
                    children=[],
                ),
                style=None,
                tags=None,
            )
            nodes.append(node)

        # Initialize the treeview with the nodes
        app.treeview.nodes = nodes

        # Create a treeview layout
        layout = app.layout.create_treeview()
        layout.add(app.treeview, 0, 0)

        # Update the dataframe in the treeview
        app.update_dataframe(df)

        # Run the application
        app.mainloop()
    except Exception as e:
        print(f"An error occurred: {e}")
        sys.exit(1)
