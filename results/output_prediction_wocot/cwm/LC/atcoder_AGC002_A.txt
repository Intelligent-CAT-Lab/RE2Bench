def predict_output():
    input_str = {
        "input": "\"1 1000000000\""
    }
    a, b = map(int, input_str["input"].strip('"').split())
    if a <= 0 and 0 <= b:
        output = 'Zero'
    elif a < 0 and min(b - a, -a) % 2 == 0:
        output = 'Negative'
    else:
        output = 'Positive'
    return {
        "output": output
    }

print(predict_output())
