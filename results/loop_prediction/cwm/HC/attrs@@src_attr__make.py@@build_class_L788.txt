def test_pre_init() -> None:
    @simple_attrs
    class C(Generic[T]):
        x: int = 42

        @T attrs.define(init=False)
        def __attrs_pre_init__(self, x: int) -> None:
            if not isinstance(x, int):
                raise TypeError("x has to be an int!")

    with raises(TypeError):
        C(x="xxx")
