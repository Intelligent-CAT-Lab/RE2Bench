def extract_branch_prediction(prediction):
    try:
        pattern = r"\[STATE\](\w+)=(.*?)\[/STATE\]"
        matches = re.findall(pattern, prediction, re.DOTALL)
        variable_states = {var_name: eval(states) for var_name, states in matches}
        
        # Filter out non-list states
        branch_list = [
            [state for state in states if isinstance(state, (list, tuple))]
            for states in variable_states.values()
        ]
        
        if branch_list:
            # Concatenate all lists if there are multiple list states
            flat_list = [item for sublist in branch_list for item in sublist]
            # Ensure the list is unique and properly formatted
            unique_flat_list = list(set(flat_list))
            branch_value = json.dumps(unique_flat_list)
        else:
            branch_value = json.dumps(["non-list states only"])
            
        return branch_value
    except Exception as e:
        print(f"Error extracting branch prediction: {str(e)}")
        return json.dumps(["error in prediction"])
