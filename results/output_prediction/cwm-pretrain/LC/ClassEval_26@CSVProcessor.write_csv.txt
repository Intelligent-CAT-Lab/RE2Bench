def quick_sort(numbers, sort_type='asc'):
    if len(numbers) <= 1:
        return numbers
    else:
        pivot = numbers[0]
        left = []
        right = []
        for i in numbers[1:]:
            if sort_type == 'asc':
                if i <= pivot:
                    left.append(i)
                else:
                    right.append(i)
            else:
                if i >= pivot:
                    left.append(i)
                else:
                    right.append(i)
        return quick_sort(left, sort_type) + [pivot] + quick_sort(right, sort_type)

    ['ME']
]

result = quick_sort(data, sort_type='desc')

class RomanNumeralConverter:
    @staticmethod
    def roman_to_int(s: str) -> int:
        roman_to_int_map = {
            'I': 1, 'V': 5, 'X': 10, 'L': 50,
            'C': 100, 'D': 500, 'M': 1000
        }

        result = 0
        prev_value = 0
        for char in s:
            current_value = roman_to_int_map[char]
            if current_value > prev_value:
                result += current_value - 2 * prev_value
            else:
                result += current_value
            prev_value = current_value

        return result

}
