import re
import string

def normalize_text(text, remove_punctuation=True):
    text = text.lower().strip()
    text = re.sub(r'\s+', ' ', text)
    if remove_punctuation:
        text = text.translate(str.maketrans('', '', string.punctuation))
    return text
