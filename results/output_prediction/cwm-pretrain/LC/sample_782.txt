import json

def evaluate_test_cases(test_cases, assistant_messages):
    # Load the expected outputs
    expected_outputs = {}
    for case in test_cases:
        question_id = case['question']['id']
        expected_outputs[question_id] = case['output']

    # Count correct answers
    correct_count = 0
    total_count = len(expected_outputs)

    # Evaluate each message
    for msg in assistant_messages:
        if msg['role'] == 'assistant':
            try:
                # Extract the JSON part from the message
                json_str = msg['content'].split('\n')[1]
                # Parse the JSON
                response = json.loads(json_str)
                # Get the question_id from the response
                question_id = response['question_id']
                # Check if the answer is correct
                if response['answer'] == expected_outputs[question_id]:
                    correct_count += 1
            except Exception as e:
                # If there's an error parsing JSON or accessing keys, treat it as incorrect
                pass

    # Calculate accuracy
    accuracy = correct_count / total_count * 100

    # Generate the results message
    result = f"Correct: {correct_count}/{total_count} ({accuracy:.2f}%)\n\n"
    for msg in assistant_messages:
        if msg['role'] == 'assistant':
            try:
                json_str = msg['content'].split('\n')[1]
                response = json.loads(json_str)
                question_id = response['question_id']
                expected = expected_outputs[question_id]
                actual = response['answer']
                if actual == expected:
                    result += f"Question {question_id}: Correct\n"
                else:
                    result += f"Question {question_id}: Incorrect\n"
            except Exception as e:
                result += f"Question {question_id}: Invalid response\n"

    return result
