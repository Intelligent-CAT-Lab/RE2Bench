import datetime
from collections import defaultdict
import statistics
from typing import List, Tuple

def number_of_cities_in_each_country(cities: List[Tuple[str, str]]) -> dict:
    result = defaultdict(int)
    for city, country in cities:
        result[country] += 1
    return dict(result)

def average_population_per_country(cities: List[Tuple[str, str]], population: List[int]) -> dict:
    country_population = defaultdict(int)
    country_count = defaultdict(int)
    for city, country in cities:
        country_population[country] += population.pop(0)
        country_count[country] += 1
    return {country: country_population[country] / country_count[country] for country in country_population}

def most_populated_country(cities: List[Tuple[str, str]], population: List[int]) -> str:
    average_pop_per_country = average_population_per_country(cities, population)
    return max(average_pop_per_country, key=average_pop_per_country.get)

def median_of_population_per_country(cities: List[Tuple[str, str]], population: List[int]) -> dict:
    country_population = defaultdict(list)
    for city, country in cities:
        country_population[country].append(population.pop(0))
    return {country: statistics.median(country_population[country]) for country in country_population}

def smallest_country_by_population(cities: List[Tuple[str, str]], population: List[int]) -> str:
    average_pop_per_country = average_population_per_country(cities, population)
    return min(average_pop_per_country, key=average_pop_per_country.get)

def largest_country_by_population(cities: List[Tuple[str, str]], population: List[int]) -> str:
    average_pop_per_country = average_population_per_country(cities, population)
    return max(average_pop_per_country, key=average_pop_per_country.get)

def number_of_countries_above_average_population(cities: List[Tuple[str, str]], population: List[int]) -> int:
    average_pop_per_country = average_population_per_country(cities, population)
    average = sum(average_pop_per_country.values()) / len(average_pop_per_country)
    return len([country for country in average_pop_per_country if average_pop_per_country[country] > average])

def oldest_country(cities: List[Tuple[str, str]], year: List[int]) -> str:
    country_age = defaultdict(list)
    for city, country in cities:
        country_age[country].append(year.pop(0))
    oldest = min(year)
    return next(country for country in country_age if oldest in country_age[country])

def youngest_country(cities: List[Tuple[str, str]], year: List[int]) -> str:
    country_age = defaultdict(list)
    for city, country in cities:
        country_age[country].append(year.pop(0))
    youngest = max(year)
    return next(country for country in country_age if youngest in country_age[country])

def number_of_people_by_birth_year(year: List[int]) -> dict:
    result = defaultdict(int)
    for y in year:
        result[y] += 1
    return dict(result)

def total_number_of_people_by_country_and_city(cities: List[Tuple[str, str]]) -> dict:
    result = defaultdict(int)
    for city, country in cities:
        result[country] += 1
    return dict(result)

def number_of_people_in_largest_country(cities: List[Tuple[str, str]], population: List[int]) -> int:
    largest_country = largest_country_by_population(cities, population)
    return sum(population) / len(population) * len(cities) / len(set(country for city, country in cities))
