from functools import lru_cache

@lru_cache(maxsize=None)
def fibonacci(n):
    if n == 0:
        return 0
    elif n == 1:
        return 1
    else:
        return fibonacci(n-1) + fibonacci(n-2)

# Define the range for which you want to calculate Fibonacci numbers
n_terms = int(input())

# Create an array to store Fibonacci numbers up to n_terms
fibonacci_numbers = [fibonacci(i) for i in range(n_terms + 1)]

# Print the Fibonacci numbers
print(fibonacci_numbers)
