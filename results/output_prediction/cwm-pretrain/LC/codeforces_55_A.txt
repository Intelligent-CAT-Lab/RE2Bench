from collections import deque
from queue import Queue
from time import perf_counter
from typing import List, Union

name = input().split(",")[0].strip()

elapsed = 0
durations = []

choice = input()

if choice == "list":
    data_struct = list(input().strip())
elif choice == "deque":
    data_struct = deque(input().strip())
elif choice == "queue":
    data_struct = Queue()
    [data_struct.put(c) for c in input().strip()]
else:
    data_struct = input().strip()

data_struct1 = data_struct.copy() if choice == "list" else data_struct

operations = int(input())

start = perf_counter()

for _ in range(operations):
    op_type, op_val = input().split(",")[:2]
    op_val = int(op_val)

    if op_type == "push":
        data_struct1.push(op_val)
    elif op_type == "pop":
        data_struct1.pop()
    elif op_type == "clear":
        data_struct1.clear()

end = perf_counter()
durations = list(map(lambda x: round(x * 1000, 3), durations))
elapsed = round((end - start) * 1000, 3)

print(f'Name: {name}')
print(f'Data Structure: {choice}')
print(f'Number of operations: {operations}')
print(f'Average time: {sum(durations) / len(durations)}ms')
print(f'Total elapsed time: {elapsed}ms')
