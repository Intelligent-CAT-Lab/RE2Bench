#include <algorithm>
using namespace std;

struct Pair {
    int x, y, w, w1;
    Pair(int x, int y, int w, int w1) : x(x), y(y), w(w), w1(w1) {}
};

bool cmp(Pair p1, Pair p2) {
    return (p1.x < p2.x) || (p1.x == p2.x && p1.y < p2.y);
}

int main() {
    int n, m;
    cin >> n >> m;

    Pair p[n + 1][n + 1];
    int mat[n + 1][n + 1];
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            p[i][j].x = i;
            p[i][j].y = j;
            cin >> mat[i][j];
            p[i][j].w = mat[i][j];
            if (mat[i][j] < i) {
                p[i][j].w1 = 0;
            } else {
                p[i][j].w1 = 1;
            }
        }
    }

    sort(p[1] + 1, p[n] + n + 1, cmp);

    int size = 1;
    int count = 1;
    int b[n + 1] = {0};
    int a[n + 1] = {0};

    for (int i = 1; i <= m; i++) {
        int x, y;
        cin >> x >> y;
        a[i] = x;
        b[i] = y;
        for (int j = 1; j < n; j++) {
            if (p[x][j].x == x && p[x][j].y > y && mat[x][p[x][j].y] == mat[x][y] && p[x][j].w1 == 0) {
                size++;
                count++;
            }
        }
    }

    cout << size << " " << count << endl;
